45 static void ssd1306_drawBufferPitch16(lcdint_t x, lcdint_t y, lcduint_t w, lcduint_t h, lcduint_t pitch, const uint8_t *data)
46 {
47  ssd1306_lcd.set_block(x, y, w);
48  while (h--)
49  {
50  lcduint_t line = w << 1;
51  while (line--)
52  {
53  ssd1306_intf.send( *data );
54  data++;
55  }
56  data += pitch - (w << 1);
57  }
58  ssd1306_intf.stop();
59 }

74 void ssd1306_drawMonoBuffer16(lcdint_t xpos, lcdint_t ypos, lcduint_t w, lcduint_t h, const uint8_t *bitmap)
75 {
76  uint8_t bit = 1;
77  uint16_t blackColor = s_ssd1306_invertByte ? ssd1306_color : 0x00;
78  uint16_t color = s_ssd1306_invertByte ? 0x00 : ssd1306_color;
79  ssd1306_lcd.set_block(xpos, ypos, w);
80  while (h--)
81  {
82  lcduint_t wx = w;
83  while (wx--)
84  {
85  uint8_t data = *bitmap;
86  if ( data & bit )
87  ssd1306_lcd.send_pixels16( color );
88  else
89  ssd1306_lcd.send_pixels16( blackColor );
90  bitmap++;
91  }
92  bit <<= 1;
93  if (bit == 0)
94  {
95  bit = 1;
96  }
97  else
98  {
99  bitmap -= w;
100  }
101  }
102  ssd1306_intf.stop();
103 }

157 void ssd1306_drawLine16(lcdint_t x1, lcdint_t y1, lcdint_t x2, lcdint_t y2)
158 {
159  lcduint_t dx = x1 > x2 ? (x1 - x2): (x2 - x1);
160  lcduint_t dy = y1 > y2 ? (y1 - y2): (y2 - y1);
161  lcduint_t err = 0;
162  if (dy > dx)
163  {
164  if (y1 > y2)
165  {
166  ssd1306_swap_data(x1, x2, lcdint_t);
167  ssd1306_swap_data(y1, y2, lcdint_t);
168  }
169  for(; y1<=y2; y1++)
170  {
171  err += dx;
172  if (err >= dy)
173  {
174  err -= dy;
175  x1 < x2 ? x1++: x1--;
176  }
177  ssd1306_putPixel16( x1, y1 );
178  }
179  }
180  else
181  {
182  if (x1 > x2)
183  {
184  ssd1306_swap_data(x1, x2, lcdint_t);
185  ssd1306_swap_data(y1, y2, lcdint_t);
186  }
187  for(; x1<=x2; x1++)
188  {
189  err += dy;
190  if (err >= dx)
191  {
192  err -= dx;
193  if (y1 < y2) y1++; else y1--;
194  }
195  ssd1306_putPixel16( x1, y1 );
196  }
197  }
198 }

300 size_t ssd1306_write16(uint8_t ch)
301 {
302  if (ch == '\r')
303  {
304  ssd1306_cursorX = 0;
305  return 0;
306  }
307  else if ( (ssd1306_cursorX > ssd1306_lcd.width - s_fixedFont.h.width) || (ch == '\n') )
308  {
309  ssd1306_cursorX = 0;
310  ssd1306_cursorY += s_fixedFont.h.height;
311  if ( ssd1306_cursorY > ssd1306_lcd.height - s_fixedFont.h.height )
312  {
313  ssd1306_cursorY = 0;
314  }
315  ssd1306_clearBlock16(0, ssd1306_cursorY, ssd1306_lcd.width, s_fixedFont.h.height);
316  if (ch == '\n')
317  {
318  return 0;
319  }
320  }
321  uint16_t unicode = ssd1306_unicode16FromUtf8(ch);
322  if (unicode == SSD1306_MORE_CHARS_REQUIRED) return 0;
323  SCharInfo char_info;
324  ssd1306_getCharBitmap(unicode, &char_info);
325  if ( 1 )
326  {
327  uint16_t color = ssd1306_color;
328  ssd1306_color = s_ssd1306_invertByte ? color : 0x0000;
329  ssd1306_fillRect16( ssd1306_cursorX, ssd1306_cursorY,
330  ssd1306_cursorX + char_info.width + char_info.spacing - 1,
331  ssd1306_cursorY + s_fixedFont.h.height - 1 );
332  ssd1306_color = color;
333  }
334  ssd1306_drawMonoBitmap16( ssd1306_cursorX,
335  ssd1306_cursorY,
336  char_info.width,
337  char_info.height,
338  char_info.glyph);
339  ssd1306_cursorX += char_info.width + char_info.spacing;
340  return 1;
341 }
