63 static void sh1106_setBlock(lcduint_t x, lcduint_t y, lcduint_t w)
64 {
65  s_column = x;
66  s_page = y;
67  ssd1306_intf.start();
68  if (ssd1306_intf.spi)
69  ssd1306_spiDataMode(0);
70  else
71  ssd1306_intf.send(0x00);
72  ssd1306_intf.send(SSD1306_SETPAGE | y);
73  ssd1306_intf.send(((x+2)>>4) | SSD1306_SETHIGHCOLUMN);
74  ssd1306_intf.send(((x+2) & 0x0f) | SSD1306_SETLOWCOLUMN);
75  if (ssd1306_intf.spi)
76  {
77  ssd1306_spiDataMode(1);
78  }
79  else
80  {
81  ssd1306_intf.stop();
82  ssd1306_intf.start();
83  ssd1306_intf.send(0x40);
84  }
85 }
